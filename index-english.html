XorLang Code Examples

note: Simple calculator program
bucket num1 = ask('Enter first number: ');
bucket num2 = ask('Enter second number: ');
bucket operation = ask('Operation (+, -, *, /): ');

when if (operation == '+') {
    bucket result = num1 + num2;
    show('Result: ' + result);
} when else when if (operation == '/') {
    bucket result = safe_divide(num1, num2);
    show('Result: ' + result);
} when else {
    error('Invalid operation');
}
note: Network and file operations
when if (is_online()) {
    show('Internet connection available');
    bucket response = http_get('https://api.example.com/data');
    file_write('response.json', response);
    show('Data saved successfully');
} when else {
    show('No internet connection');
}
note: Advanced math functions
bucket number = ask('Enter a number: ');
bucket sqrt_result = math_sqrt(number);
bucket power_result = math_pow(number, 2);
bucket random_num = math_random_range(1, 100);

show('Square root: ' + sqrt_result);
show('Squared: ' + power_result);
show('Random number: ' + random_num);

Installation

# Download installer from GitHub releases
wget https://github.com/Mr-ali-jafari/xorlang/releases/download/v3.0.1-beta/xorlang-installer-linux.zip

# Extract and install
unzip xorlang-installer-linux.zip
chmod +x install.sh
./install.sh

# Or compile from source
git clone https://github.com/Mr-ali-jafari/xorlang.git
cd xorlang
make

# Test installation
./xorlang --version
